double Equation(double x) 
{
	return pow(10-x,1/3)/*, (double)(1. / 3.))*/;
}


int main()
{
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	const double eps = 10;

	double   x0, x, xNext;
	int      nIter;

	printf("x0 =  "); cin >> x0;
	x = x0;
	xNext = Equation(x);
	nIter = 1;

	while (fabs(xNext - x) > eps)
	{
		x = xNext;
		xNext = Equation(x);
		++nIter;
		cout << "x = " << x << " xNext = " << x << " итерация " << nIter << endl;
		/*printf("%.5g %.5g %d\n", x, xNext, nIter);*/
	}
	cout << "Корень ур-я = " << xNext << " с погрешностью = " << eps << " при числе итераций = " << nIter << endl;

	return 0;

}
